  • UI: 3 Bootstrap buttons + large counter display

import React, { useState } from "react";
import "bootstrap/dist/css/bootstrap.min.css";

function App() {
  const [counter, setCounter] = useState(0);

  const increment = () => setCounter(counter + 1);
  const decrement = () => setCounter(counter - 1);
  const reset     = () => setCounter(0);

  return (
    <div className="container text-center mt-5">
      <h2 className="mb-4">WebTech Solutions Counter</h2>
      <div className="display-1 fw-bold mb-4 p-4 border rounded bg-light">
        {counter}
      </div>
      <div className="d-flex justify-content-center gap-3">
        <button className="btn btn-success btn-lg" onClick={increment}>+ Increment</button>
        <button className="btn btn-danger btn-lg"  onClick={decrement}>- Decrement</button>
        <button className="btn btn-secondary btn-lg" onClick={reset}>Reset</button>
      </div>
    </div>
  );
}

export default App;

import { useState } from "react";

function ThemeSwitcher() {
  const [isDarkMode, setIsDarkMode] = useState(false);

  const containerStyle = {
    backgroundColor: isDarkMode ? "#1a1a2e" : "#f8f9fa",
    color:           isDarkMode ? "#e2e8f0" : "#1a1a2e",
    minHeight:       "100vh",
    display:         "flex",
    flexDirection:   "column",
    alignItems:      "center",
    justifyContent:  "center",
    transition:      "all 0.3s ease",
    fontFamily:      "sans-serif",
  };

  const buttonStyle = {
    padding:      "10px 28px",
    fontSize:     "1rem",
    background:   isDarkMode ? "#a855f7" : "#6366f1",
    color:        "white",
    border:       "none",
    borderRadius: "8px",
    cursor:       "pointer",
    marginTop:    "16px",
    transition:   "background 0.3s",
  };

  return (
    <div style={containerStyle}>
      <h1>TechInnovate Solutions</h1>
      <p>Mode: {isDarkMode ? "🌙 Dark" : "☀️ Light"}</p>
      <button style={buttonStyle} onClick={() => setIsDarkMode(!isDarkMode)}>
        {isDarkMode ? "Switch to Light ☀️" : "Switch to Dark 🌙"}
      </button>
    </div>
  );
}

3app.js — Full CRUD API
📄 app.js
js 
const express = require("express");
const mysql   = require("mysql2");
const app     = express();
app.use(express.json());

const db = mysql.createConnection({
  host: "localhost", user: "root", password: "", database: "we_db"
});
db.connect(err => {
  if (err) console.error("DB Error:", err);
  else console.log("Connected to MySQL");
});

// CREATE
app.post("/products", (req, res) => {
  const { name, price, quantity } = req.body;
  db.query("INSERT INTO products (name,price,quantity) VALUES (?,?,?)",
    [name, price, quantity], (err, r) => {
    if (err) return res.status(500).json({ error: err.message });
    res.status(201).json({ message: "Product created", id: r.insertId });
  });
});

// READ ALL
app.get("/products", (req, res) => {
  db.query("SELECT * FROM products", (err, results) => {
    if (err) return res.status(500).json({ error: err.message });
    res.json(results);
  });
});

// READ ONE
app.get("/products/:id", (req, res) => {
  db.query("SELECT * FROM products WHERE id=?", [req.params.id], (err, result) => {
    if (err) return res.status(500).json({ error: err.message });
    if (!result.length) return res.status(404).json({ message: "Not found" });
    res.json(result[0]);
  });
});

// UPDATE
app.put("/products/:id", (req, res) => {
  const { name, price, quantity } = req.body;
  db.query("UPDATE products SET name=?,price=?,quantity=? WHERE id=?",
    [name, price, quantity, req.params.id], err => {
    if (err) return res.status(500).json({ error: err.message });
    res.json({ message: "Product updated" });
  });
});

// DELETE
app.delete("/products/:id", (req, res) => {
  db.query("DELETE FROM products WHERE id=?", [req.params.id], err => {
    if (err) return res.status(500).json({ error: err.message });
    res.json({ message: "Product deleted" });
  });
});

app.listen(3000, () => console.log("Server on http://localhost:3000"));

import tkinter as tk
from tkinter import messagebox

root = tk.Tk()
root.title("QuickMart Price Calculator")
root.geometry("400x300")
root.resizable(False, False)
root.configure(bg="#f0fdf4")

tk.Label(root, text="QuickMart Price Calculator",
         font=("Arial", 13, "bold"), bg="#f0fdf4", fg="#166534").pack(pady=16)

frame = tk.Frame(root, bg="#f0fdf4")
frame.pack(pady=5)

tk.Label(frame, text="Original Price ($):", font=("Arial", 10),
         bg="#f0fdf4", fg="#374151").grid(row=0, column=0, padx=14, pady=8, sticky="e")
price_entry = tk.Entry(frame, width=20, font=("Arial", 10))
price_entry.grid(row=0, column=1, padx=10)

tk.Label(frame, text="Discount (%):", font=("Arial", 10),
         bg="#f0fdf4", fg="#374151").grid(row=1, column=0, padx=14, pady=8, sticky="e")
discount_entry = tk.Entry(frame, width=20, font=("Arial", 10))
discount_entry.grid(row=1, column=1, padx=10)

result_label = tk.Label(root, text="", font=("Arial", 11, "bold"),
                        bg="#f0fdf4", fg="#15803d")
result_label.pack(pady=10)

def calculate():
    try:
        price    = float(price_entry.get())
        discount = float(discount_entry.get())
        if price <= 0:
            messagebox.showerror("Error", "Price must be greater than 0!"); return
        if not (0 <= discount <= 100):
            messagebox.showwarning("Warning", "Discount must be between 0 and 100!"); return
        amount = price * (discount / 100)
        final  = price - amount
        result_label.config(text=f"Discount: ${amount:.2f}  |  Final: ${final:.2f}")
    except ValueError:
        messagebox.showerror("Error", "Please enter valid numbers!")

def clear():
    price_entry.delete(0, "end")
    discount_entry.delete(0, "end")
    result_label.config(text="")

btn_frame = tk.Frame(root, bg="#f0fdf4")
btn_frame.pack()
tk.Button(btn_frame, text="Calculate", command=calculate,
          bg="#16a34a", fg="white", font=("Arial", 10, "bold"),
          width=14, pady=5, relief="flat", cursor="hand2").grid(row=0, column=0, padx=8)
tk.Button(btn_frame, text="Clear", command=clear,
          bg="#6b7280", fg="white", font=("Arial", 10),
          width=10, pady=5, relief="flat", cursor="hand2").grid(row=0, column=1, padx=8)

root.mainloop()

Build a car manufacturing system for "AutoTech Solutions" using the Factory Design Pattern in JavaScript — create different car types without exposing creation logic.

✅ Requirements
  • Classes: Sedan, SUV, Truck — each with getInfo()
  • Factory: CarFactory.createCar(type) returns correct object
  • Test: Create one of each, call getInfo()
💻 Solution
1carFactory.js
📄 carFactory.js
javascript
// Step 1: Car Classes
class Sedan {
  constructor() { this.type = "Sedan"; this.doors = 4; this.fuel = "Petrol"; }
  getInfo() { return `${this.type}: ${this.doors} doors, Fuel: ${this.fuel}`; }
}

class SUV {
  constructor() { this.type = "SUV"; this.doors = 5; this.fuel = "Diesel"; }
  getInfo() { return `${this.type}: ${this.doors} doors, Fuel: ${this.fuel}`; }
}

class Truck {
  constructor() { this.type = "Truck"; this.doors = 2; this.fuel = "Diesel"; }
  getInfo() { return `${this.type}: ${this.doors} doors, Fuel: ${this.fuel}`; }
}

// Step 2: Factory Class
class CarFactory {
  static createCar(type) {
    switch (type.toLowerCase()) {
      case "sedan": return new Sedan();
      case "suv":   return new SUV();
      case "truck": return new Truck();
      default: throw new Error(`Unknown car type: ${type}`);
    }
  }
}

// Step 3: Use the Factory
const sedan = CarFactory.createCar("sedan");
const suv   = CarFactory.createCar("suv");
const truck = CarFactory.createCar("truck");

console.log(sedan.getInfo()); // Sedan: 4 doors, Fuel: Petrol
console.log(suv.getInfo());   // SUV: 5 doors, Fuel: Diesel
console.log(truck.getInfo()); // Truck: 2 doors, Fuel: Diesel

Adapter PatternStructuralJS OOP
📋 Scenario

Integrate a legacy printer with a modern interface for "TechModern Solutions" using the Adapter Design Pattern.

✅ Requirements
  • LegacyPrinter: Existing class with oldPrint(text) method
  • ModernPrinter: Modern class with print(text) method
  • PrinterAdapter: Wraps LegacyPrinter, exposes print() interface
💻 Solution
1printerAdapter.js
📄 printerAdapter.js
javascript
// Step 1: Legacy Printer (old, incompatible interface)
class LegacyPrinter {
  oldPrint(text) {
    console.log(`[Legacy Printer] Printing: ${text}`);
  }
}

// Step 2: Modern Printer (expected interface)
class ModernPrinter {
  print(text) {
    console.log(`[Modern Printer] Printing: ${text}`);
  }
}

// Step 3: Adapter — bridges LegacyPrinter to ModernPrinter interface
class PrinterAdapter {
  constructor(legacyPrinter) {
    this.legacyPrinter = legacyPrinter;
  }
  print(text) {
    // Translate new interface to old interface
    this.legacyPrinter.oldPrint(text);
  }
}

// Step 4: Usage — client uses same print() interface for both
const modern  = new ModernPrinter();
modern.print("Report 2025");
// Output: [Modern Printer] Printing: Report 2025

const legacy  = new LegacyPrinter();
const adapter = new PrinterAdapter(legacy);
adapter.print("Invoice #001");
// Output: [Legacy Printer] Printing: Invoice #001

Integration TestsChained Ops
📋 Scenario

Build manual integration tests for "FinTech Solutions" using a chained Calculator class to verify multiple operations work together correctly.

💻 Solution
1calculator.js
📄 calculator.js
javascript
class Calculator {
  constructor() { this.result = 0; }
  add(n)      { this.result += n; return this; }
  subtract(n) { this.result -= n; return this; }
  multiply(n) { this.result *= n; return this; }
  divide(n)   { this.result /= n; return this; }
  getResult() { return this.result; }
  reset()     { this.result = 0; return this; }
}
module.exports = Calculator;
2integrationTest.js
📄 integrationTest.js
javascript
const Calculator = require("./calculator");

function test(description, actual, expected) {
  if (actual === expected) console.log("PASS: " + description);
  else console.log("FAIL: " + description + " | Expected: " + expected + " Got: " + actual);
}

console.log("=== FinTech Solutions - Integration Tests ===");
let calc = new Calculator();

calc.add(10).subtract(3);
test("add(10) then subtract(3) = 7", calc.getResult(), 7);

calc.reset().add(6).multiply(4).divide(3);
test("add(6), multiply(4), divide(3) = 8", calc.getResult(), 8);

calc.reset().add(10).multiply(2).subtract(5);
test("add(10), multiply(2), subtract(5) = 15", calc.getResult(), 15);

calc.reset();
test("reset() returns 0", calc.getResult(), 0);

const Calculator = require("./calculator");

describe("Calculator Integration Tests - Enterprise Finance Corp", () => {
  let calc;
  beforeEach(() => { calc = new Calculator(); });

  test("add(10) then subtract(3) = 7", () => {
    calc.add(10).subtract(3);
    expect(calc.getResult()).toBe(7);
  });
  test("add(5), multiply(4), divide(2) = 10", () => {
    calc.add(5).multiply(4).divide(2);
    expect(calc.getResult()).toBe(10);
  });
  test("add(10), multiply(2), subtract(5) = 15", () => {
    calc.add(10).multiply(2).subtract(5);
    expect(calc.getResult()).toBe(15);
  });
  test("reset() returns 0", () => {
    calc.add(100).reset();
    expect(calc.getResult()).toBe(0);
  });
  test("add(5)+add(3)+add(2) = 10", () => {
    calc.add(5).add(3).add(2);
    expect(calc.getResult()).toBe(10);
  });
});
